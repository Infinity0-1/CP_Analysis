\subsection{Exercises Series 4}
\vspace*{0.2cm}
\Large{\underline{\textbf{Exercise 1}}}
\normalsize{}\\
\noindent Study the pointwise convergence and uniform convergence of the following sequences of functions:
$$ f_n : x \longmapsto \ln\left(x + \dfrac{1}{n}\right),\ D = \R^+, \qquad
   f_n : x \longmapsto \dfrac{\ln(1 + nx)}{1 + nx},\ D = \R^+, \qquad
   f_n : x \longmapsto \dfrac{x}{n\left(1 + x^2\right)},\ D = \R. $$

\noindent\Large{\underline{\textbf{Exercise 2}}}
\normalsize{}\\
\noindent We define on $[0,1]$ the sequence of the functions $(f_n)$ by $f_n : x \longmapsto n^2x^n(1 - x^n),\ n\in\N$.
\begin{enumerate}
    \item Study the pointwise convergence of $(f_n)$ on $[0,1]$.
    \item Compare the two values bellow. What can we conclude?
        $$ \lim_{n \longrightarrow +\infty}\int_0^1 f_n(x)\ dx,\qquad \text{and } \qquad 
           \int_0^1\lim_{n \longrightarrow +\infty} f_n(x)\ dx.$$
    \item Is there a part of $[0,1]$ on which there is uniform convergence?\\
\end{enumerate}

\noindent\Large{\underline{\textbf{Exercise 3}}}
\normalsize{}\\
\begin{enumerate}
    \item Show that the sequence of functions $f_n : x \longmapsto \dfrac{x}{1 + n^2x^2}$ converges uniformly on $[-1, 1]$.
    \item Show that the sequence $(f_n')$ pointwise converges on $[-1,1]$.
    \item The sequence $(f_n')$ does it converge uniformly on $[-1,1]$?\\
\end{enumerate}

\noindent\Large{\underline{\textbf{Exercise 4}}}
\normalsize{}\\
\noindent Studying pointwise convergence, uniform convergence and normal convergence of the series of functions 
$\displaystyle\sum f_n$ on $\mathbb{R}^+$ in each of the following cases:
$$ f_n : x \longmapsto \dfrac{x}{n\left(1 + nx^2\right)}, \qquad f_n : x \longmapsto \dfrac{e^{-nx}}{1 + n^2}. $$

\noindent\Large{\underline{\textbf{Exercise 5}}}
\normalsize{}\\
\noindent Let $F$ the function defined on $]0, \pi[$ by : $F : x \longmapsto \displaystyle\sum_{n=1}^{+\infty} f_n(x)$, where
$$ f_n : x \longmapsto \dfrac{x \sin nx}{2n^{\frac{3}{2}} + \cos x}. $$
\begin{enumerate}
    \item Show that $F$ is well defined and continuous on $]0, \pi[$.
    \item Study its derivability on $]0, \pi[$.\\
\end{enumerate}

\noindent\Large{\underline{\textbf{Exercise 6}}}
\normalsize{}\\
\noindent Let the series of functions $\displaystyle\sum f_n,$ where: $\displaystyle f_n(x) = e^{-nx}\sin(2nx),\ x\in\R^+.$
\begin{enumerate}
    \item Study the pointwise convergence of this series on $\R^+$.
    \item Show that it converges uniformly on $[\alpha,+\infty],\ \forall \alpha > 0$.
    \item Show that it does not converges uniformly on $[0, \alpha],\ \forall \alpha > 0$.
\end{enumerate}
